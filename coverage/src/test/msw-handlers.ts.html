  http.get('/api/accounts', () => {

// Network delay handlers for testing loading states
export const delayedHandlers = successHandlers.map(handler => {
  const originalResponse = handler
  return http.get(handler.info.path as string, async (info) => {
    await new Promise(resolve => setTimeout(resolve, 2000))
    return originalResponse(info)
  })
})
